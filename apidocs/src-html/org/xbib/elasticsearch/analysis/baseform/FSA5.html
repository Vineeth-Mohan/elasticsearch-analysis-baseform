001package org.xbib.elasticsearch.analysis.baseform;
002
003import java.util.Collections;
004import java.util.Set;
005
006/**
007 * FSA binary format implementation for version 5.
008 * <p/>
009 * <p/>
010 * Version 5 indicates the dictionary was built with these flags:
011 * {@link FSAFlags#FLEXIBLE}, {@link FSAFlags#STOPBIT} and
012 * {@link FSAFlags#NEXTBIT}. The internal representation of the FSA must
013 * therefore follow this description (please note this format describes only a
014 * single transition (arc), not the entire dictionary file).
015 * <p/>
016 * <pre>
017 * ---- this node header present only if automaton was compiled with NUMBERS option.
018 * Byte
019 *        +-+-+-+-+-+-+-+-+\
020 *      0 | | | | | | | | | \  LSB
021 *        +-+-+-+-+-+-+-+-+  +
022 *      1 | | | | | | | | |  |      number of strings recognized
023 *        +-+-+-+-+-+-+-+-+  +----- by the automaton starting
024 *        : : : : : : : : :  |      from this node.
025 *        +-+-+-+-+-+-+-+-+  +
026 *  ctl-1 | | | | | | | | | /  MSB
027 *        +-+-+-+-+-+-+-+-+/
028 *
029 * ---- remaining part of the node
030 *
031 * Byte
032 *       +-+-+-+-+-+-+-+-+\
033 *     0 | | | | | | | | | +------ label
034 *       +-+-+-+-+-+-+-+-+/
035 *
036 *                  +------------- node pointed to is next
037 *                  | +----------- the last arc of the node
038 *                  | | +--------- the arc is final
039 *                  | | |
040 *             +-----------+
041 *             |    | | |  |
042 *         ___+___  | | |  |
043 *        /       \ | | |  |
044 *       MSB           LSB |
045 *        7 6 5 4 3 2 1 0  |
046 *       +-+-+-+-+-+-+-+-+ |
047 *     1 | | | | | | | | | \ \
048 *       +-+-+-+-+-+-+-+-+  \ \  LSB
049 *       +-+-+-+-+-+-+-+-+     +
050 *     2 | | | | | | | | |     |
051 *       +-+-+-+-+-+-+-+-+     |
052 *     3 | | | | | | | | |     +----- target node address (in bytes)
053 *       +-+-+-+-+-+-+-+-+     |      (not present except for the byte
054 *       : : : : : : : : :     |       with flags if the node pointed to
055 *       +-+-+-+-+-+-+-+-+     +       is next)
056 *   gtl | | | | | | | | |    /  MSB
057 *       +-+-+-+-+-+-+-+-+   /
058 * gtl+1                           (gtl = gotoLength)
059 * </pre>
060 */
061public final class FSA5 extends FSA {
062    /**
063     * Default filler byte.
064     */
065    public final static byte DEFAULT_FILLER = '_';
066
067    /**
068     * Default annotation byte.
069     */
070    public final static byte DEFAULT_ANNOTATION = '+';
071
072    /**
073     * Automaton version as in the file header.
074     */
075    public static final byte VERSION = 5;
076
077    /**
078     * Bit indicating that an arc corresponds to the last character of a
079     * sequence available when building the automaton.
080     */
081    public static final int BIT_FINAL_ARC = 1;
082
083    /**
084     * Bit indicating that an arc is the last one of the node's list and the
085     * following one belongs to another node.
086     */
087    public static final int BIT_LAST_ARC = 1 << 1;
088
089    /**
090     * Bit indicating that the target node of this arc follows it in the
091     * compressed automaton structure (no goto field).
092     */
093    public static final int BIT_TARGET_NEXT = 1 << 2;
094
095    /**
096     * An offset in the arc structure, where the address and flags field begins.
097     * In version 5 of FSA automata, this value is constant (1, skip label).
098     */
099    public final static int ADDRESS_OFFSET = 1;
100
101    /**
102     * An array of bytes with the internal representation of the automaton.
103     * Please see the documentation of this class for more information on how
104     * this structure is organized.
105     */
106    public byte[] arcs;
107
108    /**
109     * The length of the node header structure (if the automaton was compiled with
110     * <code>NUMBERS</code> option). Otherwise zero.
111     */
112    public int nodeDataLength;
113
114    /**
115     * Flags for this automaton version.
116     */
117    private Set<FSAFlags> flags;
118
119    /**
120     * Number of bytes each address takes in full, expanded form (goto length).
121     */
122    public int gtl;
123
124    /**
125     * Filler character.
126     */
127    public byte filler;
128
129    /**
130     * Annotation character.
131     */
132    public byte annotation;
133
134    public FSA5() {
135    }
136
137    public FSA5 setArcs(byte[] arcs) {
138        this.arcs = arcs;
139        return this;
140    }
141
142    public FSA5 setNodeDataLength(int nodeDataLength) {
143        this.nodeDataLength = nodeDataLength;
144        return this;
145    }
146
147    public FSA5 setFlags(Set<FSAFlags> flags) {
148        this.flags = flags;
149        return this;
150    }
151
152    public FSA5 setGotoLength(int gtl) {
153        this.gtl = gtl;
154        return this;
155    }
156
157    public FSA5 setFiller(byte filler) {
158        this.filler = filler;
159        return this;
160    }
161
162    public FSA5 setAnnotation(byte annotation) {
163        this.annotation = annotation;
164        return this;
165    }
166
167    /**
168     * Returns the start node of this automaton.
169     */
170    @Override
171    public int getRootNode() {
172        // Skip dummy node marking terminating state.
173        final int epsilonNode = skipArc(getFirstArc(0));
174        // And follow the epsilon node's first (and only) arc.
175        return getDestinationNodeOffset(getFirstArc(epsilonNode));
176    }
177
178    /**
179     * {@inheritDoc}
180     */
181    @Override
182    public final int getFirstArc(int node) {
183        return nodeDataLength + node;
184    }
185
186    /**
187     * {@inheritDoc}
188     */
189    @Override
190    public final int getNextArc(int arc) {
191        if (isArcLast(arc)) {
192            return 0;
193        } else {
194            return skipArc(arc);
195        }
196    }
197
198    /**
199     * {@inheritDoc}
200     */
201    @Override
202    public int getArc(int node, byte label) {
203        for (int arc = getFirstArc(node); arc != 0; arc = getNextArc(arc)) {
204            if (getArcLabel(arc) == label) {
205                return arc;
206            }
207        }
208
209        // An arc labeled with "label" not found.
210        return 0;
211    }
212
213    /**
214     * {@inheritDoc}
215     */
216    @Override
217    public int getEndNode(int arc) {
218        final int nodeOffset = getDestinationNodeOffset(arc);
219        assert nodeOffset != 0 : "No target node for terminal arcs.";
220        return nodeOffset;
221    }
222
223    /**
224     * {@inheritDoc}
225     */
226    @Override
227    public byte getArcLabel(int arc) {
228        return arcs[arc];
229    }
230
231    /**
232     * {@inheritDoc}
233     */
234    @Override
235    public boolean isArcFinal(int arc) {
236        return (arcs[arc + ADDRESS_OFFSET] & BIT_FINAL_ARC) != 0;
237    }
238
239    /**
240     * {@inheritDoc}
241     */
242    @Override
243    public boolean isArcTerminal(int arc) {
244        return (0 == getDestinationNodeOffset(arc));
245    }
246
247    /**
248     * Returns the number encoded at the given node. The number equals the count
249     * of the set of suffixes reachable from <code>node</code> (called its right
250     * language).
251     */
252    @Override
253    public int getRightLanguageCount(int node) {
254        assert getFlags().contains(FSAFlags.NUMBERS) : "This FSA was not compiled with NUMBERS.";
255        return decodeFromBytes(arcs, node, nodeDataLength);
256    }
257
258    /**
259     * {@inheritDoc}
260     * <p/>
261     * <p>For this automaton version, an additional {@link FSAFlags#NUMBERS} flag
262     * may be set to indicate the automaton contains extra fields for each node.</p>
263     */
264    @Override
265    public Set<FSAFlags> getFlags() {
266        return Collections.unmodifiableSet(flags);
267    }
268
269    /**
270     * Returns <code>true</code> if this arc has <code>LAST</code> bit set.
271     *
272     * @see #BIT_LAST_ARC
273     */
274    public boolean isArcLast(int arc) {
275        return (arcs[arc + ADDRESS_OFFSET] & BIT_LAST_ARC) != 0;
276    }
277
278    /**
279     * @see #BIT_TARGET_NEXT
280     */
281    public boolean isNextSet(int arc) {
282        return (arcs[arc + ADDRESS_OFFSET] & BIT_TARGET_NEXT) != 0;
283    }
284
285    /**
286     * Returns an n-byte integer encoded in byte-packed representation.
287     */
288    static int decodeFromBytes(final byte[] arcs, final int start, final int n) {
289        int r = 0;
290        for (int i = n; --i >= 0; ) {
291            r = r << 8 | (arcs[start + i] & 0xff);
292        }
293        return r;
294    }
295
296    /**
297     * Returns the address of the node pointed to by this arc.
298     */
299    final int getDestinationNodeOffset(int arc) {
300        if (isNextSet(arc)) {
301                        /* The destination node follows this arc in the array. */
302            return skipArc(arc);
303        } else {
304                        /*
305                         * The destination node address has to be extracted from the arc's
306                         * goto field.
307                         */
308            return decodeFromBytes(arcs, arc + ADDRESS_OFFSET, gtl) >>> 3;
309        }
310    }
311
312    /**
313     * Read the arc's layout and skip as many bytes, as needed.
314     */
315    private int skipArc(int offset) {
316        return offset + (isNextSet(offset)
317                ? 1 + 1 /* label + flags */
318                : 1 + gtl /* label + flags/address */);
319    }
320}
