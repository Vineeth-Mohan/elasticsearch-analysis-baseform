001package org.xbib.elasticsearch.analysis.baseform;
002
003import java.util.EnumSet;
004import java.util.Set;
005
006/**
007 * CFSA (Compact Finite State Automaton) binary format implementation, version 2:
008 * <ul>
009 * <li>{@link #BIT_TARGET_NEXT} applicable on all arcs, not necessarily the last one.</li>
010 * <li>v-coded goto field</li>
011 * <li>v-coded perfect hashing numbers, if any</li>
012 * <li>31 most frequent labels integrated with flags byte</li>
013 * </ul>
014 * <p/>
015 * <p>The encoding of automaton body is as follows.</p>
016 * <p/>
017 * <pre>
018 * ---- CFSA header
019 * Byte                            Description
020 *       +-+-+-+-+-+-+-+-+\
021 *     0 | | | | | | | | | +------ '\'
022 *       +-+-+-+-+-+-+-+-+/
023 *       +-+-+-+-+-+-+-+-+\
024 *     1 | | | | | | | | | +------ 'f'
025 *       +-+-+-+-+-+-+-+-+/
026 *       +-+-+-+-+-+-+-+-+\
027 *     2 | | | | | | | | | +------ 's'
028 *       +-+-+-+-+-+-+-+-+/
029 *       +-+-+-+-+-+-+-+-+\
030 *     3 | | | | | | | | | +------ 'a'
031 *       +-+-+-+-+-+-+-+-+/
032 *       +-+-+-+-+-+-+-+-+\
033 *     4 | | | | | | | | | +------ version (fixed 0xc6)
034 *       +-+-+-+-+-+-+-+-+/
035 *       +-+-+-+-+-+-+-+-+\
036 *     5 | | | | | | | | | +----\
037 *       +-+-+-+-+-+-+-+-+/      \ flags [MSB first]
038 *       +-+-+-+-+-+-+-+-+\      /
039 *     6 | | | | | | | | | +----/
040 *       +-+-+-+-+-+-+-+-+/
041 *       +-+-+-+-+-+-+-+-+\
042 *     7 | | | | | | | | | +------ label lookup table size
043 *       +-+-+-+-+-+-+-+-+/
044 *       +-+-+-+-+-+-+-+-+\
045 *  8-32 | | | | | | | | | +------ label value lookup table
046 *       : : : : : : : : : |
047 *       +-+-+-+-+-+-+-+-+/
048 *
049 * ---- Start of a node; only if automaton was compiled with NUMBERS option.
050 *
051 * Byte
052 *        +-+-+-+-+-+-+-+-+\
053 *      0 | | | | | | | | | \
054 *        +-+-+-+-+-+-+-+-+  +
055 *      1 | | | | | | | | |  |      number of strings recognized
056 *        +-+-+-+-+-+-+-+-+  +----- by the automaton starting
057 *        : : : : : : : : :  |      from this node. v-coding
058 *        +-+-+-+-+-+-+-+-+  +
059 *        | | | | | | | | | /
060 *        +-+-+-+-+-+-+-+-+/
061 *
062 * ---- A vector of this node's arcs. An arc's layout depends on the combination of flags.
063 *
064 * 1) NEXT bit set, mapped arc label.
065 *
066 *        +----------------------- node pointed to is next
067 *        | +--------------------- the last arc of the node
068 *        | | +------------------- this arc leads to a final state (acceptor)
069 *        | | |  _______+--------- arc's label; indexed if M > 0, otherwise explicit label follows
070 *        | | | / | | | |
071 *       +-+-+-+-+-+-+-+-+\
072 *     0 |N|L|F|M|M|M|M|M| +------ flags + (M) index of the mapped label.
073 *       +-+-+-+-+-+-+-+-+/
074 *       +-+-+-+-+-+-+-+-+\
075 *     1 | | | | | | | | | +------ optional label if M == 0
076 *       +-+-+-+-+-+-+-+-+/
077 *       : : : : : : : : :
078 *       +-+-+-+-+-+-+-+-+\
079 *       |A|A|A|A|A|A|A|A| +------ v-coded goto address
080 *       +-+-+-+-+-+-+-+-+/
081 * </pre>
082 */
083public final class CFSA2 extends FSA {
084    /**
085     * Automaton header version value.
086     */
087    public static final byte VERSION = (byte) 0xC6;
088
089    /**
090     * The target node of this arc follows the last arc of the current state
091     * (no goto field).
092     */
093    public static final int BIT_TARGET_NEXT = 1 << 7;
094
095    /**
096     * The arc is the last one from the current node's arcs list.
097     */
098    public static final int BIT_LAST_ARC = 1 << 6;
099
100    /**
101     * The arc corresponds to the last character of a sequence
102     * available when building the automaton (acceptor transition).
103     */
104    public static final int BIT_FINAL_ARC = 1 << 5;
105
106    /**
107     * The count of bits assigned to storing an indexed label.
108     */
109    public static final int LABEL_INDEX_BITS = 5;
110
111    /**
112     * Masks only the M bits of a flag byte.
113     */
114    public static final int LABEL_INDEX_MASK = (1 << LABEL_INDEX_BITS) - 1;
115
116    /**
117     * Maximum size of the labels index.
118     */
119    public static final int LABEL_INDEX_SIZE = (1 << LABEL_INDEX_BITS) - 1;
120
121    /**
122     * An array of bytes with the internal representation of the automaton.
123     * Please see the documentation of this class for more information on how
124     * this structure is organized.
125     */
126    public byte[] arcs;
127
128    /**
129     * Flags for this automaton version.
130     */
131    private EnumSet<FSAFlags> flags;
132
133    /**
134     * Label mapping for M-indexed labels.
135     */
136    public byte[] labelMapping;
137
138    /**
139     * If <code>true</code> states are prepended with numbers.
140     */
141    private boolean hasNumbers;
142
143    public CFSA2() {
144    }
145
146    public CFSA2 setArcs(byte[] arcs) {
147        this.arcs = arcs;
148        return this;
149    }
150
151    public CFSA2 hasNumbers(boolean hasNumbers) {
152        this.hasNumbers = hasNumbers;
153        return this;
154    }
155
156    public CFSA2 setLabelMapping(byte[] labelMapping) {
157        this.labelMapping = labelMapping;
158        return this;
159    }
160
161    @Override
162    public int getRootNode() {
163        int epsilon = 0;
164        return getDestinationNodeOffset(getFirstArc(epsilon));
165    }
166
167    /**
168     * {@inheritDoc}
169     */
170    @Override
171    public final int getFirstArc(int node) {
172        if (hasNumbers) {
173            return skipVInt(node);
174        } else {
175            return node;
176        }
177    }
178
179    /**
180     * {@inheritDoc}
181     */
182    @Override
183    public final int getNextArc(int arc) {
184        if (isArcLast(arc)) {
185            return 0;
186        } else {
187            return skipArc(arc);
188        }
189    }
190
191    /**
192     * {@inheritDoc}
193     */
194    @Override
195    public int getArc(int node, byte label) {
196        for (int arc = getFirstArc(node); arc != 0; arc = getNextArc(arc)) {
197            if (getArcLabel(arc) == label) {
198                return arc;
199            }
200        }
201        // An arc labeled with "label" not found.
202        return 0;
203    }
204
205    /**
206     * {@inheritDoc}
207     */
208    @Override
209    public int getEndNode(int arc) {
210        final int nodeOffset = getDestinationNodeOffset(arc);
211        assert nodeOffset != 0 : "can't follow a terminal arc: " + arc;
212        assert nodeOffset < arcs.length : "node out of bounds";
213        return nodeOffset;
214    }
215
216    /**
217     * {@inheritDoc}
218     */
219    @Override
220    public byte getArcLabel(int arc) {
221        int index = arcs[arc] & LABEL_INDEX_MASK;
222        if (index > 0) {
223            return this.labelMapping[index];
224        } else {
225            return arcs[arc + 1];
226        }
227    }
228
229    /**
230     * {@inheritDoc}
231     */
232    @Override
233    public int getRightLanguageCount(int node) {
234        assert getFlags().contains(FSAFlags.NUMBERS) : "this FSA was not compiled with NUMBERS";
235        return readVInt(arcs, node);
236    }
237
238    /**
239     * {@inheritDoc}
240     */
241    @Override
242    public boolean isArcFinal(int arc) {
243        return (arcs[arc] & BIT_FINAL_ARC) != 0;
244    }
245
246    /**
247     * {@inheritDoc}
248     */
249    @Override
250    public boolean isArcTerminal(int arc) {
251        return (0 == getDestinationNodeOffset(arc));
252    }
253
254    /**
255     * Returns <code>true</code> if this arc has <code>NEXT</code> bit set.
256     *
257     * @see #BIT_LAST_ARC
258     */
259    public boolean isArcLast(int arc) {
260        return (arcs[arc] & BIT_LAST_ARC) != 0;
261    }
262
263    /**
264     * @see #BIT_TARGET_NEXT
265     */
266    public boolean isNextSet(int arc) {
267        return (arcs[arc] & BIT_TARGET_NEXT) != 0;
268    }
269
270    /**
271     * {@inheritDoc}
272     */
273    public Set<FSAFlags> getFlags() {
274        return flags;
275    }
276
277    /**
278     * Returns the address of the node pointed to by this arc.
279     */
280    public final int getDestinationNodeOffset(int arc) {
281        if (isNextSet(arc)) {
282                    /* Follow until the last arc of this state. */
283            while (!isArcLast(arc)) {
284                arc = getNextArc(arc);
285            }
286                        /* And return the byte right after it. */
287            return skipArc(arc);
288        } else {
289                        /*
290                         * The destination node address is v-coded. v-code starts either
291                         * at the next byte (label indexed) or after the next byte (label explicit).
292                         */
293            return readVInt(arcs, arc + ((arcs[arc] & LABEL_INDEX_MASK) == 0 ? 2 : 1));
294        }
295    }
296
297    /**
298     * Read the arc's layout and skip as many bytes, as needed, to skip it.
299     */
300    private int skipArc(int offset) {
301        int flag = arcs[offset++];
302        // Explicit label?
303        if ((flag & LABEL_INDEX_MASK) == 0) {
304            offset++;
305        }
306        // Explicit goto?
307        if ((flag & BIT_TARGET_NEXT) == 0) {
308            offset = skipVInt(offset);
309        }
310        assert offset < this.arcs.length;
311        return offset;
312    }
313
314    /**
315     * Read a v-int.
316     */
317    public static int readVInt(byte[] array, int offset) {
318        byte b = array[offset];
319        int value = b & 0x7F;
320        for (int shift = 7; b < 0; shift += 7) {
321            b = array[++offset];
322            value |= (b & 0x7F) << shift;
323        }
324        return value;
325    }
326
327    /**
328     * Write a v-int to a byte array.
329     */
330    public static int writeVInt(byte[] array, int offset, int value) {
331        assert value >= 0 : "can't v-code negative ints";
332        while (value > 0x7F) {
333            array[offset++] = (byte) (0x80 | (value & 0x7F));
334            value >>= 7;
335        }
336        array[offset++] = (byte) value;
337        return offset;
338    }
339
340    /**
341     * Return the byte-length of a v-coded int.
342     */
343    public static int vIntLength(int value) {
344        assert value >= 0 : "can't v-code negative ints";
345        int bytes;
346        for (bytes = 1; value >= 0x80; bytes++) {
347            value >>= 7;
348        }
349        return bytes;
350    }
351
352    /**
353     * Skip a v-int.
354     */
355    private int skipVInt(int offset) {
356        while (arcs[offset++] < 0) {
357            ;
358        }
359        return offset;
360    }
361}
