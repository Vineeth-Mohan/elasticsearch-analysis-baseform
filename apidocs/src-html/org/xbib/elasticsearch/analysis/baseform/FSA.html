001    package org.xbib.elasticsearch.analysis.baseform;
002    
003    import java.nio.ByteBuffer;
004    import java.util.BitSet;
005    import java.util.Collections;
006    import java.util.Iterator;
007    import java.util.Set;
008    
009    /**
010     * This is an abstract class for handling finite state automata. These
011     * automata are arc-based, a design described in Jan Daciuk's <i>Incremental
012     * Construction of Finite-State Automata and Transducers, and Their Use in the
013     * Natural Language Processing</i> (PhD thesis, Technical University of Gdansk).
014     * <p/>
015     * <p>
016     * Concrete subclasses (implementations) provide varying tradeoffs and features:
017     * traversal speed vs. memory size, for example.
018     * </p>
019     *
020     * @see FSABuilder
021     */
022    public abstract class FSA implements Iterable<ByteBuffer> {
023        /**
024         * @return Returns the identifier of the root node of this automaton.
025         *         Returns 0 if the start node is also the end node (the automaton
026         *         is empty).
027         */
028        public abstract int getRootNode();
029    
030        /**
031         * @return Returns the identifier of the first arc leaving <code>node</code>
032         *         or 0 if the node has no outgoing arcs.
033         */
034        public abstract int getFirstArc(int node);
035    
036        /**
037         * @return Returns the identifier of the next arc after <code>arc</code> and
038         *         leaving <code>node</code>. Zero is returned if no more arcs are
039         *         available for the node.
040         */
041        public abstract int getNextArc(int arc);
042    
043        /**
044         * @return Returns the identifier of an arc leaving <code>node</code> and
045         *         labeled with <code>label</code>. An identifier equal to 0 means
046         *         the node has no outgoing arc labeled <code>label</code>.
047         */
048        public abstract int getArc(int node, byte label);
049    
050        /**
051         * Return the label associated with a given <code>arc</code>.
052         */
053        public abstract byte getArcLabel(int arc);
054    
055        /**
056         * Returns <code>true</code> if the destination node at the end of this
057         * <code>arc</code> corresponds to an input sequence created when building
058         * this automaton.
059         */
060        public abstract boolean isArcFinal(int arc);
061    
062        /**
063         * Returns <code>true</code> if this <code>arc</code> does not have a
064         * terminating node (@link {@link #getEndNode(int)} will throw an
065         * exception). Implies {@link #isArcFinal(int)}.
066         */
067        public abstract boolean isArcTerminal(int arc);
068    
069        /**
070         * Return the end node pointed to by a given <code>arc</code>. Terminal arcs
071         * (those that point to a terminal state) have no end node representation
072         * and throw a runtime exception.
073         */
074        public abstract int getEndNode(int arc);
075    
076        /**
077         * Returns a set of flags for this FSA instance.
078         */
079        public abstract Set<FSAFlags> getFlags();
080    
081        /**
082         * @return Returns the number of sequences reachable from the given state if
083         *         the automaton was compiled with {@link FSAFlags#NUMBERS}. The size of
084         *         the right language of the state, in other words.
085         * @throws UnsupportedOperationException If the automaton was not compiled with
086         *                                       {@link FSAFlags#NUMBERS}. The value can then be computed by manual count
087         *                                       of {@link #getSequences(int)}.
088         */
089        public int getRightLanguageCount(int node) {
090            throw new UnsupportedOperationException("Automaton not compiled with " + FSAFlags.NUMBERS);
091        }
092    
093        /**
094         * Returns an iterator over all binary sequences starting at the given FSA
095         * state (node) and ending in final nodes. This corresponds to a set of
096         * suffixes of a given prefix from all sequences stored in the automaton.
097         * <p/>
098         * <p>
099         * The returned iterator is a {@link java.nio.ByteBuffer} whose contents changes on
100         * each call to {@link java.util.Iterator#next()}. The keep the contents between calls
101         * to {@link java.util.Iterator#next()}, one must copy the buffer to some other
102         * location.
103         * </p>
104         * <p/>
105         * <p>
106         * <b>Important.</b> It is guaranteed that the returned byte buffer is
107         * backed by a byte array and that the content of the byte buffer starts at
108         * the array's index 0.
109         * </p>
110         *
111         * @see Iterable
112         */
113        public Iterable<ByteBuffer> getSequences(final int node) {
114            if (node == 0) {
115                return Collections.<ByteBuffer>emptyList();
116            }
117            return new Iterable<ByteBuffer>() {
118                public Iterator<ByteBuffer> iterator() {
119                    return new FSAFinalStatesIterator(FSA.this, node);
120                }
121            };
122        }
123    
124        /**
125         * An alias of calling {@link #iterator} directly ({@link FSA} is also
126         * {@link Iterable}).
127         */
128        public final Iterable<ByteBuffer> getSequences() {
129            return getSequences(getRootNode());
130        }
131    
132        /**
133         * Returns an iterator over all binary sequences starting from the initial
134         * FSA state (node) and ending in final nodes. The returned iterator is a
135         * {@link java.nio.ByteBuffer} whose contents changes on each call to
136         * {@link java.util.Iterator#next()}. The keep the contents between calls to
137         * {@link java.util.Iterator#next()}, one must copy the buffer to some other location.
138         * <p/>
139         * <p>
140         * <b>Important.</b> It is guaranteed that the returned byte buffer is
141         * backed by a byte array and that the content of the byte buffer starts at
142         * the array's index 0.
143         * </p>
144         *
145         * @see Iterable
146         */
147        public final Iterator<ByteBuffer> iterator() {
148            return getSequences().iterator();
149        }
150    
151        /**
152         * Visit all states. The order of visiting is undefined. This method may be faster
153         * than traversing the automaton in post or preorder since it can scan states
154         * linearly. Returning false from {@link StateVisitor#accept(int)}
155         * immediately terminates the traversal.
156         */
157        public <T extends StateVisitor> T visitAllStates(T v) {
158            return visitInPostOrder(v);
159        }
160    
161        /**
162         * Same as {@link #visitInPostOrder(StateVisitor, int)},
163         * starting from root automaton node.
164         */
165        public <T extends StateVisitor> T visitInPostOrder(T v) {
166            return visitInPostOrder(v, getRootNode());
167        }
168    
169        /**
170         * Visits all states reachable from <code>node</code> in postorder.
171         * Returning false from {@link StateVisitor#accept(int)}
172         * immediately terminates the traversal.
173         */
174        public <T extends StateVisitor> T visitInPostOrder(T v, int node) {
175            visitInPostOrder(v, node, new BitSet());
176            return v;
177        }
178    
179        /**
180         * Private recursion.
181         */
182        private boolean visitInPostOrder(StateVisitor v, int node, BitSet visited) {
183            if (visited.get(node)) {
184                return true;
185            }
186            visited.set(node);
187            for (int arc = getFirstArc(node); arc != 0; arc = getNextArc(arc)) {
188                if (!isArcTerminal(arc)) {
189                    if (!visitInPostOrder(v, getEndNode(arc), visited)) {
190                        return false;
191                    }
192                }
193            }
194            return v.accept(node);
195        }
196    
197        /**
198         * Same as {@link #visitInPreOrder(StateVisitor, int)}, starting from root automaton node.
199         */
200        public <T extends StateVisitor> T visitInPreOrder(T v) {
201            return visitInPreOrder(v, getRootNode());
202        }
203    
204        /**
205         * Visits all states in preorder. Returning false from {@link StateVisitor#accept(int)}
206         * skips traversal of all sub-states of a given state.
207         */
208        public <T extends StateVisitor> T visitInPreOrder(T v, int node) {
209            visitInPreOrder(v, node, new BitSet());
210            return v;
211        }
212    
213        /**
214         * Private recursion.
215         */
216        private void visitInPreOrder(StateVisitor v, int node, BitSet visited) {
217            if (visited.get(node)) {
218                return;
219            }
220            visited.set(node);
221            if (v.accept(node)) {
222                for (int arc = getFirstArc(node); arc != 0; arc = getNextArc(arc)) {
223                    if (!isArcTerminal(arc)) {
224                        visitInPreOrder(v, getEndNode(arc), visited);
225                    }
226                }
227            }
228        }
229    
230    
231    }
