001    package org.xbib.elasticsearch.index.analysis;
002    
003    import java.io.IOException;
004    import java.io.InputStreamReader;
005    
006    import org.apache.lucene.analysis.TokenStream;
007    import org.elasticsearch.ElasticsearchIllegalArgumentException;
008    import org.elasticsearch.common.inject.Inject;
009    import org.elasticsearch.common.inject.assistedinject.Assisted;
010    import org.elasticsearch.common.settings.Settings;
011    import org.elasticsearch.env.Environment;
012    import org.elasticsearch.index.Index;
013    import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
014    import org.elasticsearch.index.settings.IndexSettings;
015    import org.xbib.elasticsearch.analysis.baseform.Dictionary;
016    
017    public class BaseformTokenFilterFactory extends AbstractTokenFilterFactory {
018    
019        private final Dictionary dictionary;
020    
021        @Inject
022        public BaseformTokenFilterFactory(Index index,
023                                            @IndexSettings Settings indexSettings, Environment env,
024                                            @Assisted String name, @Assisted Settings settings) {
025            super(index, indexSettings, name, settings);
026            this.dictionary = createDictionary(env, settings);
027        }
028    
029        @Override
030        public TokenStream create(TokenStream tokenStream) {
031            return new BaseformTokenFilter(tokenStream, dictionary);
032        }
033    
034        private Dictionary createDictionary(Environment env, Settings settings) {
035            try {
036                String lang = settings.get("language", "de");
037                String path = "/" + lang + "-lemma-utf8.txt";
038                return new Dictionary().load(new InputStreamReader(env.resolveConfig(path).openStream(), "UTF-8"));
039            } catch (IOException e) {
040                throw new ElasticsearchIllegalArgumentException("resources in settings not found: " + settings, e);
041            }
042        }
043    }
